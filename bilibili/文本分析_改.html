library(topicmodels)
## Warning: 程序包'topicmodels'是用R版本4.4.2 来建造的
# 分批处理数据，每次处理1000条文档（主题分析构建矩阵时不分批内存会爆）
batch_size <- 1000
num_batches <- ceiling(length(seg_filtered) / batch_size)

# 确定全局词汇表
global_vocab <- character()

for (i in 1:num_batches) {
  start_idx <- (i - 1) * batch_size + 1
  end_idx <- min(i * batch_size, length(seg_filtered))
  
  # 将分词结果转换为文档 - 词矩阵（DTM）
  corpus <- Corpus(VectorSource(seg_filtered[start_idx:end_idx]))
  dtm_batch <- DocumentTermMatrix(corpus)
  
  # 计算每个词的词频
  word_freqs <- colSums(as.matrix(dtm_batch))
  
  # 筛选出高频词的索引
  high_freq_indices <- which(word_freqs > 5)
  
  # 获取高频词
  high_freq_words <- names(word_freqs)[high_freq_indices]
  
  # 更新全局词汇表
  global_vocab <- union(global_vocab, high_freq_words)
  
  # 释放内存
  rm(corpus, dtm_batch)
  gc()
}

dtm_list <- list()

for (i in 1:num_batches) {
  start_idx <- (i - 1) * batch_size + 1
  end_idx <- min(i * batch_size, length(seg_filtered))
  
  # 将分词结果转换为文档 - 词矩阵（DTM）
  corpus <- Corpus(VectorSource(seg_filtered[start_idx:end_idx]))
  dtm_batch <- DocumentTermMatrix(corpus, control = list(dictionary = global_vocab))
  
  dtm_list[[i]] <- dtm_batch
  
  # 释放内存
  rm(corpus, dtm_batch)
  gc()
}

# 合并所有批次的DTM
dtm <- do.call(rbind, dtm_list)

# 过滤掉所有元素都为零的行（避免出现0元素的行）
non_zero_rows <- rowSums(as.matrix(dtm)) > 0
dtm <- dtm[non_zero_rows, ]

# 构建LDA模型，设置主题数量为5
lda_model <- LDA(dtm, k = 5, control = list(seed = 1234))
